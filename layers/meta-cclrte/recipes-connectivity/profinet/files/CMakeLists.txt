cmake_minimum_required(VERSION 3.14)
project(pnet VERSION 0.1.0 LANGUAGES C)

# ── Options (reasonable defaults for a single-port PROFINET device) ──────────
set(PNET_OPTION_FAST_STARTUP      0)
set(PNET_OPTION_PARAMETER_SERVER  0)
set(PNET_OPTION_IR                0)
set(PNET_OPTION_SR                0)
set(PNET_OPTION_REDUNDANCY        0)
set(PNET_OPTION_AR_VENDOR_BLOCKS  0)
set(PNET_OPTION_RS                0)
set(PNET_OPTION_MC_CR             0)
set(PNET_OPTION_SRL               0)
set(PNET_OPTION_SNMP              0)
set(PNET_OPTION_DRIVER_ENABLE     0)
set(PNET_USE_ATOMICS              1)

set(PNET_MAX_AR                   2)
set(PNET_MAX_API                  1)
set(PNET_MAX_CR                   2)
set(PNET_MAX_SLOTS                5)
set(PNET_MAX_SUBSLOTS             3)
set(PNET_MAX_DFP_IOCR             0)
set(PNET_MAX_PHYSICAL_PORTS       1)
set(PNET_MAX_LOG_BOOK_ENTRIES     16)
set(PNET_MAX_ALARMS               3)
set(PNET_MAX_ALARM_PAYLOAD_DATA_SIZE 28)
set(PNET_MAX_DIAG_ITEMS           100)
set(PNET_MAX_DIAG_MANUF_DATA_SIZE 8)
set(PNET_MAX_MC_CR                1)
set(PNET_MAX_AR_VENDOR_BLOCKS     1)
set(PNET_MAX_AR_VENDOR_BLOCK_DATA_LENGTH 32)
set(PNET_MAX_MAN_SPECIFIC_FAST_STARTUP_DATA_LENGTH 0)
set(PNET_MAX_SESSION_BUFFER_SIZE  4096)
set(PNET_MAX_DIRECTORYPATH_SIZE   240)
set(PNET_MAX_FILENAME_SIZE        24)
set(PNET_MAX_PORT_DESCRIPTION_SIZE 64)

set(LOG_LEVEL     "WARNING")
set(PF_ETH_LOG    "OFF")
set(PF_LLDP_LOG   "OFF")
set(PF_SNMP_LOG   "OFF")
set(PF_CPM_LOG    "OFF")
set(PF_PPM_LOG    "OFF")
set(PF_DCP_LOG    "OFF")
set(PF_RPC_LOG    "OFF")
set(PF_ALARM_LOG  "OFF")
set(PF_AL_BUF_LOG "OFF")
set(PF_PNAL_LOG   "OFF")
set(PNET_LOG      "OFF")

set(PNET_VERSION_MAJOR  0)
set(PNET_VERSION_MINOR  1)
set(PNET_VERSION_PATCH  0)
set(PNET_GIT_REVISION   "public")

# ── Generate headers ─────────────────────────────────────────────────────────
configure_file(pnet_options.h.in pnet_options.h)
configure_file(pnet_version.h.in pnet_version.h)

# ── Sources ───────────────────────────────────────────────────────────────────
file(GLOB_RECURSE PNET_SOURCES
    src/common/*.c
    src/device/*.c
)
file(GLOB OSAL_SOURCES osal/src/linux/*.c)

# ── Library ───────────────────────────────────────────────────────────────────
add_library(pnet SHARED ${PNET_SOURCES} ${OSAL_SOURCES})

# pnet_api.h includes pnet_export.h — generated by CMake's export header mechanism
include(GenerateExportHeader)
generate_export_header(pnet
    BASE_NAME pnet
    EXPORT_FILE_NAME ${CMAKE_CURRENT_BINARY_DIR}/pnet_export.h
)

target_include_directories(pnet
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/osal/include>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}>
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src
        ${CMAKE_CURRENT_SOURCE_DIR}/src/common
        ${CMAKE_CURRENT_SOURCE_DIR}/src/device
        ${CMAKE_CURRENT_SOURCE_DIR}/osal/src/linux
)

target_compile_options(pnet PRIVATE -Wall -Wextra -Wno-unused-parameter)

# ── Install ───────────────────────────────────────────────────────────────────
include(GNUInstallDirs)
install(TARGETS pnet
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)
install(FILES
    include/pnet_api.h
    include/pnal.h
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)
install(FILES
    ${CMAKE_CURRENT_BINARY_DIR}/pnet_options.h
    ${CMAKE_CURRENT_BINARY_DIR}/pnet_version.h
    ${CMAKE_CURRENT_BINARY_DIR}/pnet_export.h
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)
